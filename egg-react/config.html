# webpack.config.js

easywebpack@4.8.0 开始支持，因为有了默认配置，所以最新的骨架项目中，webpack.config.js 文件为非必须配置。

使用 node-glob 遍历文件。下面配置会自动遍历 app/web/page 目录的所有 .jsx 文件作为 entry 入口，排除 component|components|view|views 目录下的文件。 这个是 egg react ssr 项目默认配置, 同时使用 react-entry-template-loader 作为模板入口。 注意：只有 entry 文件是 .jsx 文件(非.js）时，才会自动使用 react-entry-template-loader 模板。

# wepback entry  jsx 文件入口(支持前端渲染和服务端渲染）

这种模式直接支持 React Component 作为 webpack entry, 无需编写 React.render 初始化代码，统一 react-entry-template-loader 插件提供的模板。注意：以下配置为 easywebpack 默认配置。

// webpack.config.js
module.exports = {
 // 注意 只有 entry 文件是 .jsx 文件(非.js）时，才会自动使用 react-entry-template-loader 模板
 entry: 'app/web/page/**!(component|components|view|views)/*.jsx'
}

# 前端渲染 wepback entry  js 文件入口

// ${root}/app/web/page/app/index.js
import React from 'react';
import ReactDom from 'react-dom';
import { AppContainer } from 'react-hot-loader';
import Entry from './index.jsx';
const state = window.__INITIAL_STATE__;
ReactDom.render(EASY_ENV_IS_DEV ? <appcontainer><app {...state}></app></appcontainer> : <app {...state}>, document.getElementById('app'));
// local development mode, hot reload
if (EASY_ENV_IS_DEV && module.hot) {
 module.hot.accept();
}

# 服务端渲染 wepback entry  js 文件入口

import React, { Component } from 'react'
import ReactDOM from 'react-dom'
import { Provider } from 'react-redux'
import {match, RouterContext} from 'react-router'
import { BrowserRouter, StaticRouter } from 'react-router-dom';
import { matchRoutes, renderRoutes } from 'react-router-config';
import { AppContainer } from 'react-hot-loader';
import Layout from '../../component/layout';
import App from './components';
import { create } from './components/store';
import routes from './components/router'
import './index.css';

const clientRender = () => {
 const store = create(window.__INITIAL_STATE__);
 const url = store.getState().url;
 const Entry = () => (<div>
   <provider store="{" }>
     <browserrouter>
       <app url="{" }>
     </app></browserrouter>
   </provider>
 </div>
 );
 const render = Page =>{
   ReactDOM.hydrate(EASY_ENV_IS_DEV ? <appcontainer><page></page></appcontainer> : <page>, document.getElementById('app'));
 };
 if (EASY_ENV_IS_DEV && module.hot) {
   module.hot.accept();
 }
 render(Entry);
};

const serverRender = (context, options)=> {
 const url = context.state.url;
 const branch = matchRoutes(routes, url);
 const promises = branch.map(({route}) => {
   const fetch = route.component.fetch;
   return fetch instanceof Function ? fetch() : Promise.resolve(null)
 });
 return Promise.all(promises).then(data => {
   const initState = context.state;
   data.forEach(item => {
     Object.assign(initState, item);
   });
   context.state = Object.assign({}, context.state, initState);
   const store = create(initState);
   return () =>(
     <layout>
       <div>
         <provider store="{store}">
           <staticrouter location="{url}" context="{{}}">
             <app url="{url}/">
           </app></staticrouter>
         </provider>
       </div>
     </layout>
   )
 });
};
export default EASY_ENV_IS_NODE ?  serverRender : clientRender();
